	SOV_relocate_industry_from_moscow = {
		icon = generic_construction
		highlight_states = {
			state = 219
		}	
		available = {
			NOT = {
				219 = { #Moscow
					is_owned_and_controlled_by = SOV
					OR = {
						arms_factory > 0
						industrial_complex > 0
					}
				} 
			}
			
		}

		is_good = yes

		allowed = {
			always = no
			#added through on_action
		}

		fire_only_once = no
		days_mission_timeout = 7

		visible = {
			SOV = { has_war_with = GER }
		}

        timeout_effect = {
				if = {
					limit = { 
						219 = { #Moscow
							is_owned_and_controlled_by = SOV
							arms_factory > 0
						} 
					}
					219 = { #Moscow
						remove_building = {
							type = arms_factory
							level = 1
						}
					}
					572	= {
						add_extra_state_shared_building_slots = 1										
						add_building_construction = {
							type = arms_factory
							level = 1
							instant_build = yes
						}
						hidden_effect = {
							damage_building = { type = arms_factory damage = 0.2 }
						}						
					}
					hidden_effect = {
						if = {
							limit = { 
								OR = {
									219 = { #Moscow
										is_owned_and_controlled_by = SOV
										arms_factory > 0
									}
									219 = { #Moscow
										is_owned_and_controlled_by = SOV
										industrial_complex > 0
									} 									
								}	
							}
							activate_mission = SOV_relocate_industry_from_moscow
